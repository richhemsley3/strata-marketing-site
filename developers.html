123456789101112import strata

# Initialize the client
client = strata.Client(api_key="sk_live_...")

# Connect a data source
source = client.sources.create(
    type="postgresql",
    connection_string="postgres://user:pass@host/db"
)

# Run a classification scan
scan = client.classifications.run(source_id=source.id)
print(scan.results)

Go 
12345678910111213141516package main

import "github.com/strata-io/strata-go"

func main() {
    // Initialize the client
    client := strata.NewClient("sk_live_...")

    // Connect a data source
    source, _ := client.Sources.Create(&strata.SourceParams{
        Type:             "postgresql",
        ConnectionString: "postgres://user:pass@host/db",
    })

    // Run a classification scan
    scan, _ := client.Classifications.Run(source.ID)
    fmt.Println(scan.Results)
}
