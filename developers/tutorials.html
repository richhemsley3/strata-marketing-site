QUICKSTARTS

Start with your stack.

1

Install

Terminal 
pip install strata-sdk
2

Configure

Python 
import os
from strata import StrataClient

client = StrataClient(
    api_key=os.environ["STRATA_API_KEY"]
)
3

Run your first scan

Python 
scan = client.scans.create(
    source_id="src_pg_01",
    classify=True
)

# Poll until complete
result = scan.wait()
print(f"Classified {result.fields_scanned} fields")
print(f"Found {result.sensitive_count} sensitive")
1

Install

Terminal 
npm install @strata/sdk
2

Configure

JavaScript 
import { StrataClient } from '@strata/sdk';

const client = new StrataClient({
  apiKey: process.env.STRATA_API_KEY
});
3

Run your first scan

JavaScript 
const scan = await client.scans.create({
  sourceId: 'src_pg_01',
  classify: true
});

const result = await scan.waitForCompletion();
console.log(`Classified ${result.fieldsScanned} fields`);
console.log(`Found ${result.sensitiveCount} sensitive`);
1

Install

Terminal 
go get github.com/strata-io/strata-go
2

Configure

Go 
import (
    "os"
    strata "github.com/strata-io/strata-go"
)

client := strata.NewClient(
    strata.WithAPIKey(os.Getenv("STRATA_API_KEY")),
)
3

Run your first scan

Go 
scan, err := client.Scans.Create(ctx, &strata.ScanParams{
    SourceID: "src_pg_01",
    Classify: true,
})

result, err := scan.Wait(ctx)
fmt.Printf("Classified %d fields\n", result.FieldsScanned)
fmt.Printf("Found %d sensitive\n", result.SensitiveCount)
1

Install

Maven 
<dependency>
  <groupId>io.strata</groupId>
  <artifactId>strata-sdk</artifactId>
  <version>1.0.0</version>
</dependency>
2

Configure

Java 
import io.strata.sdk.StrataClient;

StrataClient client = StrataClient.builder()
    .apiKey(System.getenv("STRATA_API_KEY"))
    .build();
3

Run your first scan

Java 
Scan scan = client.scans().create(
    ScanParams.builder()
        .sourceId("src_pg_01")
        .classify(true)
        .build()
);

ScanResult result = scan.waitForCompletion();
System.out.printf("Classified %d fields%n", result.getFieldsScanned());
System.out.printf("Found %d sensitive%n", result.getSensitiveCount());
